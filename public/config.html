            const [geminiKey, setGeminiKey] = useState('');

                }

            };

                                    className="w-full p-3 bg-slate-700 text-white rounded-lg border border-slate-600 focus:outline-none focus:ring-2 focus:ring-blue-500 transition"

                                />

                            </div>

                            <div>

                                <label className="block text-sm font-medium text-slate-300 mb-2">

                                    TMDB API Key

                                </label>

                                <input

                                    type="text"

                                    value={tmdbKey}

                                    onChange={(e) => setTmdbKey(e.target.value)}

                                    placeholder="Enter TMDB API Key"

                                    className="w-full p-3 bg-slate-700 text-white rounded-lg border border-slate-600 focus:outline-none focus:ring-2 focus:ring-blue-500 transition"

                            <h3 className="text-sm font-medium text-slate-300 mb-3">Example Searches</h3>

                            <pre className="text-xs text-blue-400 bg-slate-800 p-3 rounded-lg">

                                "A heartwarming comedy about family relationships"<br />

                                "Action comedy with martial arts"

                        </div>

                    </div>

                </div>

        }

    </script>
